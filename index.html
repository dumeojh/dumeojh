// 🔗 1. 본인의 구글 시트 CSV URL을 정확히 넣으세요. // 형식 예시: https://docs.google.com/spreadsheets/d/e/XXXXX/pub?output=csv const SHEET_CSV_URL = "여기에_복사한_링크를_넣으세요"; let db = []; let curGroupIdx = null; async function fetchSheetData() { try { const response = await fetch(SHEET_CSV_URL); if (!response.ok) throw new Error("네트워크 응답에 문제가 있습니다."); const data = await response.text(); // CSV 줄바꿈 처리 (윈도우/맥/리눅스 공통 대응) const rows = data.split(/\r?\n/); const groupedData = {}; // 2열(인덱스 1)부터 데이터 시작 for (let i = 1; i < rows.length; i++) { if (!rows[i].trim()) continue; // 빈 줄 건너뛰기 // 쉼표로 분리 (데이터 내부에 쉼표가 있을 경우를 대비한 정규식) const cols = rows[i].split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/); if (cols.length >= 3) { const groupName = cols[0].replace(/"/g, "").trim(); // A열: 교과군 const subjectName = cols[1].replace(/"/g, "").trim(); // B열: 과목명 const fileName = cols[2].replace(/"/g, "").trim(); // C열: 파일명 if (!groupName || !subjectName) continue; if (!groupedData[groupName]) { groupedData[groupName] = { name: groupName, subjects: [] }; } // PDF 폴더 경로와 합침 groupedData[groupName].subjects.push({ n: subjectName, url: "pdfs/" + fileName }); } } db = Object.values(groupedData); if (db.length === 0) { document.getElementById('loading').innerText = "불러올 데이터가 없습니다. 시트를 확인해주세요."; } else { renderGroups(); document.getElementById('loading').style.display = 'none'; } } catch (error) { console.error("데이터 로드 실패:", error); document.getElementById('loading').innerText = "에러 발생: " + error.message; } }
